export const createStyleSheet = () => { const styleElement = document.createElement('style') document.head.appendChild(styleElement) return styleElement.sheet } export const getRule = (styleSheet, selector) => { for (const rule of styleSheet.cssRules) { if (rule.selectorText === selector) { return rule } } return null } export const createRule = (styleSheet, rule) => { const index = styleSheet.insertRule(rule, styleSheet.cssRules.length) return styleSheet.cssRules[index] } export default { createStyleSheet, getRule, createRule }
