export const ASC = 'asc' export const DESC = 'desc' export const NONE = 'none' export const encode = ({ order, name }) => { switch (order) { case ASC: return `+${name}` case DESC: return `-${name}` default: return name } } export const decode = ordering => { const [, sort, name] = /(\+|-)?(.+)/.exec(ordering) let order switch (sort) { case '+': order = ASC break case '-': order = DESC break default: order = NONE break } return { order, name } }
